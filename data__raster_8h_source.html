159 #pragma omp parallel for

164  }
165  }

1118 template <typename T, typename MASK_T>
1119 clsRasterData<T, MASK_T>::clsRasterData(vector<string>& filenames,
1120  const bool calc_pos /* = true */,
1121  clsRasterData<MASK_T>* mask /* = nullptr */,
1122  const bool use_mask_ext /* = true */,
1123  T default_value /* = static_cast<T>(NODATA_VALUE) */,
1124  STRING_MAP opts /* = STRING_MAP() */) {
1125  InitializeRasterClass();
1126  if (filenames.empty()) {
1127  // if filenames is empty
1128  StatusNoUnitTest("The filenames MUST have at least one raster file path!");
1129  return;
1130  }
1131  if (!CheckRasterFilesExist(filenames)) {
1132  // if not all filenames are presented
1133  StatusNoUnitTest("Please make sure all file path existed!");
1134  return;
1135  }
1136  if (filenames.size() == 1) {
1137  // if filenames has only one file
1138  if (!ConstructFromSingleFile(filenames[0], calc_pos, mask,
1139  use_mask_ext, default_value, opts)) {
1140  return;
1141  }
1142  } else {
1143  // construct from multi-layers file
1144  n_lyrs_ = CVT_INT(filenames.size());
1145  // 1. firstly, take the first layer as the main input, to calculate position index or
1146  // extract by mask if stated.
1147  if (!ConstructFromSingleFile(filenames[0], calc_pos, mask,
1148  use_mask_ext, default_value, opts)) {
1149  return;
1150  }
1151  // 2. then, change the core file name and file path template which format is: `<file dir>/CoreName_%d.<suffix>`
1152  core_name_ = SplitString(core_name_, '_').at(0);
1153  full_path_ = GetPathFromFullName(filenames[0]) + SEP + core_name_ + "_%d." + GetSuffix(filenames[0]);
1154  // So, to get a given layer's filepath, please use the following code. Definitely, maximum 99 layers is supported now.
1155  // string layerFilepath = m_filePathName.replace(m_filePathName.find_last_of("%d") - 1, 2, ValueToString(1));
1156  // 3. initialize raster_2d_ and read the other layers according to position data if stated,
1157  // or just read by row and col
1158  Initialize2DArray(n_cells_, n_lyrs_, raster_2d_, no_data_value_);
1159 #pragma omp parallel for
1160  for (int i = 0; i < n_cells_; i++) {
1161  raster_2d_[i][0] = raster_[i];
1162  }
1163  Release1DArray(raster_);
1164  // take the first layer as mask, and use_mask_ext is true, and no need to calculate position data
1165  for (int fileidx = 1; fileidx < CVT_INT(filenames.size()); fileidx++) {
1166  map<string, double> tmpheader;
1167  T* tmplyrdata = nullptr;
1168  string curfilename = filenames[fileidx];
1169  if (StringMatch(GetUpper(GetSuffix(curfilename)), string(ASCIIExtension))) {
1170  ReadAscFile(curfilename, &tmpheader, &tmplyrdata);
1171  } else {
1172 #ifdef USE_GDAL
1173  ReadRasterFileByGdal(curfilename, &tmpheader, &tmplyrdata);
1174 #else
1175  cout << "Warning: Only ASC format is supported without GDAL!" << endl;
1176  return;
1177 #endif
1178  }
1179  if (calc_pos_) {
1180 #pragma omp parallel for
1181  for (int i = 0; i < n_cells_; ++i) {
1182  int tmp_row = raster_pos_data_[i][0];
1183  int tmp_col = raster_pos_data_[i][1];
1184  AddOtherLayerRasterData(tmp_row, tmp_col, i, fileidx, tmpheader, tmplyrdata);
1185  }
1186  } else {
1187 #pragma omp parallel for
1188  for (int i = 0; i < CVT_INT(headers_.at(HEADER_RS_NROWS)); i++) {
