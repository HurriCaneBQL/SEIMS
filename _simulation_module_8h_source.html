199 /*!
200  * Macros for CheckInputData function
201  * BE REMEMBER OF SEMICOLON!
202  */
203 //! CHECK_DATA is used for the unforeseen situation
204 #define CHECK_DATA(moduleID, expression, desc) if ((expression)) \
205  throw ModelException(moduleID, "CheckInputData", string(desc))
206 //! CHECK_POINTER is used for 1D or 2D raster and other pointer of data
207 #define CHECK_POINTER(moduleID, param) if (nullptr == (param)) \
208  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST NOT be NULL!"))
209 //! CHECK_POSITIVE is used for single value that must be positive
210 #define CHECK_POSITIVE(moduleID, param) if ((param) <= 0) \
211  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST be positive!"))
212 //! CHECK_NONNEGATIVE is used for single value that must be greater or equal than zero
213 #define CHECK_NONNEGATIVE(moduleID, param) if ((param) < 0) \
214  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST be greater or equal than zero!"))
215 //! CHECK_NEGATIVE is used for single value that must be negative
216 #define CHECK_NEGATIVE(moduleID, param) if ((param) >= 0) \
217  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST be negative!"))
218 //! CHECK_ZERO is used for single value that must not be ZERO
219 #define CHECK_ZERO(moduleID, param) if ((param) == 0 || FloatEqual(CVT_FLT(param), 0.f)) \
220  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST NOT be zero!"))
221 //! CHECK_NODATA is used for single value that must not be NODATA_VALUE
222 #define CHECK_NODATA(moduleID, param) if ((param) == NODATA_VALUE || FloatEqual(CVT_FLT(param), NODATA_VALUE)) \
223  throw ModelException(moduleID, "CheckInputData", string(#param) + string(" MUST NOT be NODATA_VALUE!"))
